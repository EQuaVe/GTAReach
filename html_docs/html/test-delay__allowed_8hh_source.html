23 TEST_CASE("delay allowed for all reference clocks", "[delay_allowed]")
24 {
25  std::string model = "system:delay_allowed \n\
26  \n\
27  process:P1 \n\
28  location:P1:l0{initial:} \n\
29  \n\
30  process:P2 \n\
31  location:P2:l0{initial:} \n\
32  \n\
33  process:P3 \n\
34  location:P3:l0{initial:} \n\
35  ";
36 
37  std::unique_ptr<tchecker::parsing::system_declaration_t const> sysdecl{tchecker::test::parse(model)};
38  assert(sysdecl != nullptr);
39 
40  tchecker::ta::system_t system{*sysdecl};
41 
42  tchecker::process_id_t const P1 = system.process_id("P1");
43  tchecker::process_id_t const P2 = system.process_id("P2");
44  tchecker::process_id_t const P3 = system.process_id("P3");
45 
46  tchecker::loc_id_t const P1_l0 = system.location(P1, "l0")->id();
47  tchecker::loc_id_t const P2_l0 = system.location(P2, "l0")->id();
48  tchecker::loc_id_t const P3_l0 = system.location(P3, "l0")->id();
49 
50  tchecker::vloc_t * vloc = tchecker::vloc_allocate_and_construct(system.processes_count(), system.processes_count());
51  (*vloc)[P1] = P1_l0;
52  (*vloc)[P2] = P2_l0;
53  (*vloc)[P3] = P3_l0;
54 
55  SECTION("Synchronous delay allowed") { REQUIRE(tchecker::ta::delay_allowed(system, *vloc)); }
56 
57  SECTION("Asynchronous delay allowed, single reference clock")
58  {
59  tchecker::reference_clock_variables_t r =
60  single_reference_clocks(system.clock_variables().flattened(), system.processes_count());
61 
62  boost::dynamic_bitset<> delay_allowed = tchecker::ta::delay_allowed(system, r, *vloc);
63 
64  REQUIRE(delay_allowed.all());
65  }
66 
67  SECTION("Asynchronous delay allowed, per process reference clock")
68  {
69  tchecker::variable_access_map_t va_map = tchecker::variable_access(system);
70 
71  tchecker::reference_clock_variables_t r =
72  process_reference_clocks(va_map, system.clock_variables().flattened(), system.processes_count());
73 
74  boost::dynamic_bitset<> delay_allowed = tchecker::ta::delay_allowed(system, r, *vloc);
75 
76  REQUIRE(delay_allowed.all());
77  }
78 
79  tchecker::vloc_destruct_and_deallocate(vloc);
80 }
