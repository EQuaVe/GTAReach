124  template <class... ARGS> tchecker::intrusive_shared_ptr_t<T> construct(ARGS &&... args)
125  {
126  void * t = allocate();
127  if (t == nullptr)
128  return tchecker::intrusive_shared_ptr_t<T>(nullptr);
129  // p points after the reference counter

151  bool destruct(tchecker::intrusive_shared_ptr_t<T> & p)
152  {
153  assert(p->refcount() >= 1);
154 
155  if (p.ptr() == nullptr)
156  return false;
157  if (p->refcount() > 1)
158  return false;
159 
160  T * t = p.ptr();
161  T::destruct(t);
162 
163  typename T::refcount_t * chunk = reinterpret_cast<typename T::refcount_t *>(t) - 1;
164  release(chunk);
165 
166  p = nullptr;

178  std::size_t collect()
179  {
180  std::size_t collected = 0;
181  void *collected_begin = nullptr, *collected_end = nullptr;
182 
183  // Move to the free list all chunks in blocks list that are not in the
184  // free list and that are unused
185  for (void * block = _block_head; block != nullptr; block = nextblock(block)) {
186  void * block_end = static_cast<char *>(block) + _block_size;
187 
188  for (char * chunk = first_chunk_ptr(block); chunk != block_end; chunk += _alloc_size) {
189  // Ignore chunks inside unused raw block (refcount not set yet)
190  if ((_raw_head <= chunk) && (chunk < _raw_end))
191  break; // ignore the entire raw block
192 
193  // Collect all unused chunks in the free list
194  typename T::refcount_t * refcount = reinterpret_cast<typename T::refcount_t *>(chunk);
195  if (*refcount == COLLECTABLE_CHUNK) {
196  // make the chunk free using its refcount
197  *refcount = FREE_CHUNK;
198  // destruct the object in the chunk
199  T * t = reinterpret_cast<T *>(refcount + 1);
200  T::destruct(t); // t->~T()
201  // add the chunk to the list of collected chunks
202  this->nextchunk(chunk) = collected_begin;
203  if (collected_end == nullptr)
204  collected_end = chunk;
205  collected_begin = chunk;
206  ++collected;
207  }
208  }
209  }
210 
211  // release the collected chunks
212  if (collected > 0)
213  this->release(collected_begin, collected_end);
214 
215  return collected;
216  }

226  // - are not in free list

228  for (void * block = _block_head; block != nullptr; block = nextblock(block)) {
