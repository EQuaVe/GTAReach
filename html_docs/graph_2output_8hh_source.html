84 template <class GRAPH, class NODE_LE, class EDGE_LE>
85 std::ostream & dot_output(std::ostream & os, GRAPH const & g, std::string const & name)
86 {
87  using node_id_t = std::size_t;
88  using extended_edge_t = std::tuple<node_id_t, node_id_t, typename GRAPH::edge_sptr_t>; // <src, tgt, edge_sptr>
89 
90  // Extend EDGE_LE on triples (src, tgt, edge)
91  class extended_edge_le_t : private EDGE_LE {
92  public:
93  bool operator()(extended_edge_t const & e1, extended_edge_t const & e2) const
94  {
95  auto && [src1, tgt1, edge_sptr1] = e1;
96  auto && [src2, tgt2, edge_sptr2] = e2;
97  if (src1 < src2)
98  return true;
99  else if (src1 > src2)
100  return false;
101  else if (tgt1 < tgt2)
102  return true;
103  else if (tgt1 > tgt2)
104  return false;
105  else
106  return EDGE_LE::operator()(edge_sptr1, edge_sptr2);
107  }
108  };
109 
110  // Sort nodes and given them an ID
111  std::map<typename GRAPH::node_sptr_t, node_id_t, NODE_LE> nodes_map;
112  for (typename GRAPH::node_sptr_t const & n : g.nodes())
113  nodes_map.insert(std::make_pair(n, 0));
114 
115  node_id_t count = 0;
116  for (auto & [node, id] : nodes_map) {
117  id = count;
118  ++count;
119  }
120 
121  // Sort (extended) edges
122  std::multiset<extended_edge_t, extended_edge_le_t> edges_set;
123  for (typename GRAPH::node_sptr_t const & n : g.nodes())
124  for (typename GRAPH::edge_sptr_t const & e : g.outgoing_edges(n)) {
125  node_id_t src = nodes_map[g.edge_src(e)];
126  node_id_t tgt = nodes_map[g.edge_tgt(e)];
127  edges_set.insert(std::make_tuple(src, tgt, e));
128  }
129 
130  // output graph
131  std::map<std::string, std::string> attr;
132 
133  tchecker::graph::dot_output_header(os, name);
134 
135  for (auto && [node, id] : nodes_map) {
136  attr.clear();
137  g.attributes(node, attr);
138  tchecker::graph::dot_output_node(os, std::to_string(id), attr);
139  }
140 
141  for (auto && [src, tgt, edge] : edges_set) {
142  attr.clear();
143  g.attributes(edge, attr);
144  tchecker::graph::dot_output_edge(os, std::to_string(src), std::to_string(tgt), attr);
145  }
146 
147  tchecker::graph::dot_output_footer(os);
148 
149  return os;
150 }
