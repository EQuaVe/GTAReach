27 /* IMPLEMENTATION NOTES:
28  Difference bound #c is encoded in a 32-bits integer with the least-significant bit used to store the
29  comparator #: LE (<=) or LT (<), and the other bits used to store c.
30 
31  Comparison with "cleaner" implementations, e.g.:
32  struct db_t { struct db_t {
33  unsigned cmp : 1; int32_t db; // managed as db_t below
34  int value : 31; };
35  };
36  have shown that current implementation is the fastest. This is due to using a primitive non-composite
37  type (no constructor), and to arithmetic and bit-level operations.
38  */

122 inline tchecker::dbm::db_t sum(tchecker::dbm::db_t db1, tchecker::dbm::db_t db2)
123 {
124  if ((db1 == tchecker::dbm::LT_INFINITY) || (db2 == tchecker::dbm::LT_INFINITY))
125  return tchecker::dbm::LT_INFINITY;
126 #if defined(DBM_UNSAFE)
127  return ((db1 + db2) - ((db1 | db2) & 1));
128 #else
129  return tchecker::dbm::db(((db1 & tchecker::dbm::LE) && (db2 & tchecker::dbm::LE)) ? tchecker::dbm::LE : tchecker::dbm::LT,
130  (db1 >> 1) + (db2 >> 1));
131 #endif // DBM_UNSAFE
132 }
