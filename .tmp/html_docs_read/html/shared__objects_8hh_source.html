156  inline void take_reference(void) const
157  {
158  refcount_t * refcount = refcount_addr();
159  *refcount += 1;
160  if (*refcount == REFCOUNT_MAX) // overflow

162  }

170  inline void release_reference(void) const
171  {
172  refcount_t * refcount = refcount_addr();
173  if (*refcount == 0)
174  throw std::underflow_error("reference counter underflow");
175  *refcount -= 1;
176  }
