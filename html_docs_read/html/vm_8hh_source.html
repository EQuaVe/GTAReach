32 enum instruction_t {
33  // assume stack = v1 ... vK and top symbol is vK
34  VM_RET = 0, // end of operation, return vK
35  VM_RETZ, // end of operation when vK==0, return vK
36  VM_FAILNOTIN, // raise exception when (l <= vK <= h) does not hold, for parameters l and h of VM_FAILNOTIN
37  //
38  VM_JMP, // unconditional jump relatively to next instruction;
39  // offset is a parameter of the instruction
40  VM_JMPZ, // stack = v1 ... vK jump if vK == 0
41  // offset is a parameter of the instruction
42  VM_PUSH, // stack = v1 ... vK v where v is a parameter of VM_PUSH
43  //
44  VM_VALUEAT, // stack = v1 ... vK-1 [vK] vK replaced by value at addr vK
45  VM_ASSIGN, // stack = v1 ... vK-2 [vK-1] = vK, i.e. value at address vK-1 is replaced by vK
46  //
47  VM_LAND, // stack = v1 ... vK-2 (vK-1 && vK)
48  VM_MINUS, // stack = v1 ... vK-2 (vK-1 - vK)
49  VM_DIV, // stack = v1 ... vK-2 (vK-1 / vK)
50  VM_EQ, // stack = v1 ... vK-2 (vK-1 == vK)
51  VM_GE, // stack = v1 ... vK-2 (vK-1 >= vK)
52  VM_GT, // stack = v1 ... vK-2 (vK-1 > vK)
53  VM_LT, // stack = v1 ... vK-2 (vK-1 < vK)
54  VM_LE, // stack = v1 ... vK-2 (vK-1 <= vK)
55  VM_MUL, // stack = v1 ... vK-2 (vK-1 * vK)
56  VM_MOD, // stack = v1 ... vK-2 (vK-1 % vK)
57  VM_NE, // stack = v1 ... vK-2 (vK-1 != vK)
58  VM_SUM, // stack = v1 ... vK-2 (vK-1 + vK)
59  VM_NEG, // stack = v1 ... vK-1 (- vK)
60  VM_LNOT, // stack = v1 ... vK-1 (! vK)
61  //
62  VM_CLKCONSTR, // stack = v1 ... vK-4 1 output (vK-2 vK-1 s vK) where
63  // s is a parameter of VM_CLKCONSTR (strictness)
64  VM_CLKRESET, // stack = v1 ... vK-3 output (vK-2 vK-1 vK)
65  //
66  VM_PUSH_FRAME, // push a new frame for local variables
67  VM_POP_FRAME, // pop the top-level frame
68  VM_VALUEAT_FRAME, // stack = v1 ... vK-1 [vK]
69  // vK is replaced by the value of the local variable identified by vK.
70  VM_ASSIGN_FRAME, // stack = v1 ... vK-2
71  // [vK-1] is assigned vK where vK-1 identifies a local variables.
72  VM_INIT_FRAME, // stack = v1 ... vK-2
73  // [vK-1] is initialized with vK where vK-1 identifies a local variables.
74  VM_NOP, // SHOULD BE LAST INSTRUCTION
75 };

227  inline tchecker::integer_t interpret_instruction(tchecker::bytecode_t const *& bytecode,
228  tchecker::intvars_valuation_t & intval,
229  tchecker::clock_constraint_container_t & clkconstr,
230  tchecker::clock_reset_container_t & clkreset)
231  {
232  // Assume stack=v1 ... vK where vK is the top symbol
233  switch (*bytecode) {
234  // end of operation, return vK
235  case VM_RET: {
236  auto val = top_and_pop<tchecker::integer_t>();
237  assert(size() == 0);
238  clear();
239  _return = true;
240  return val;
241  }
242 
243  // end of operation when vK==0, return 0
244  case VM_RETZ: {
245  if (top<tchecker::integer_t>() == 0) {
246  clear();
247  _return = true;
248  return 0;
249  }
250  return top<tchecker::integer_t>();
251  }
252 
253  // raise exception when not (l <= vK <= h) for parameters l and h
254  // of instruction VM_FAILNOTIN
255  case VM_FAILNOTIN: {
256  tchecker::bytecode_t const l = *++bytecode;
257  tchecker::bytecode_t const h = *++bytecode;
258  auto const offset = top<tchecker::bytecode_t>();
259  assert(contains_value<tchecker::integer_t>(l));
260  assert(contains_value<tchecker::integer_t>(h));
261  assert(contains_value<tchecker::integer_t>(offset));
262  if ((offset < l) || (offset > h)) {
263  std::stringstream ss;
264  ss << offset << " out of [" << l << ", " << h << "]";
265  throw std::out_of_range("out-of-bounds value: " + ss.str());
266  }
267 
268  return top<tchecker::integer_t>();
269  }
270 
271  // unconditional jump relatively to next instruction;
272  // offset is a parameter of the instruction
273  case VM_JMP: {
274  tchecker::bytecode_t const shift = *++bytecode;
275  // jump is relative to the address of the next instruction:
276  // - bytecode++;
277  // but we have to handle the increment of the IP in the main loop:
278  // - bytecode += shift - 1;
279  // so:
280  bytecode += shift;
281 
282  return 1;
283  }
284  // stack = v1 ... vK jump if vK == 0
285  // offset is a parameter of the instruction
286  case VM_JMPZ: {
287  tchecker::bytecode_t const shift = *++bytecode;
288 
289  if (top_and_pop<tchecker::integer_t>() == 0) {
290  bytecode += shift;
291  return 0;
292  }
293  return 1;
294  }
295 
296  // stack = v1 ... vK v where v is a parameter of instruction VM_PUSH
297  case VM_PUSH: {
298  tchecker::bytecode_t const v = *++bytecode;
299  push<tchecker::bytecode_t>(v);
300  return top<tchecker::integer_t>();
301  }
302 
303  // stack = v1 ... [vK] vK replaced by value at ID vK in intvars
304  // valuation
305  case VM_VALUEAT: {
306  auto const id = top_and_pop<tchecker::intvar_id_t>();
307  assert(id < intval.size());
308  push<tchecker::integer_t>(intval[id]);
309  return top<tchecker::integer_t>();
310  }
311 
312  // [vK-1] = vK, stack = v1 ... vK-2
313  case VM_ASSIGN: {
314  auto const value = top_and_pop<tchecker::integer_t>();
315  auto const id = top_and_pop<tchecker::intvar_id_t>();
316  assert(id < intval.size());
317  intval[id] = value;
318  return value;
319  }
320 
321  // stack = v1 ... vK-2 (vK-1 && vK)
322  case VM_LAND: {
323  auto const right = top_and_pop<tchecker::integer_t>();
324  auto const left = top_and_pop<tchecker::integer_t>();
325  push<tchecker::integer_t>(left && right);
326  return top<tchecker::integer_t>();
327  }
328 
329  // stack = v1 ... vK-2 (vK-1 - vK)
330  case VM_MINUS: {
331  auto const right = top_and_pop<tchecker::integer_t>();
332  auto const left = top_and_pop<tchecker::integer_t>();
333  push<tchecker::integer_t>(left - right);
334  return top<tchecker::integer_t>();
335  }
336 
337  // stack = v1 ... vK-2 (vK-1 / vK)
338  case VM_DIV: {
339  auto const right = top_and_pop<tchecker::integer_t>();
340  auto const left = top_and_pop<tchecker::integer_t>();
341  push<tchecker::integer_t>(left / right);
342  return top<tchecker::integer_t>();
343  }

345  // stack = v1 ... vK-2 (vK-1 == vK)
346  case VM_EQ: {
347  auto const right = top_and_pop<tchecker::integer_t>();
348  auto const left = top_and_pop<tchecker::integer_t>();
349  push<tchecker::integer_t>(left == right);
350  return top<tchecker::integer_t>();
351  }

354  case VM_GE: {
355  auto const right = top_and_pop<tchecker::integer_t>();
356  auto const left = top_and_pop<tchecker::integer_t>();
357  push<tchecker::integer_t>(left >= right);
358  return top<tchecker::integer_t>();
359  }

361  // stack = v1 ... vK-2 (vK-1 > vK)
362  case VM_GT: {
363  auto const right = top_and_pop<tchecker::integer_t>();
364  auto const left = top_and_pop<tchecker::integer_t>();
365  push<tchecker::integer_t>(left > right);
366  return top<tchecker::integer_t>();
367  }
368

370  case VM_LT: {
371  auto const right = top_and_pop<tchecker::integer_t>();
372  auto const left = top_and_pop<tchecker::integer_t>();
373  push<tchecker::integer_t>(left < right);
374  return top<tchecker::integer_t>();
375  }

380  auto const left = top_and_pop<tchecker::integer_t>();
381  push<tchecker::integer_t>(left <= right);
382  return top<tchecker::integer_t>();
383  }
384 
385  // stack = v1 ... vK-2 (vK-1 * vK)
386  case VM_MUL: {
387  auto const right = top_and_pop<tchecker::integer_t>();
388  auto const left = top_and_pop<tchecker::integer_t>();
389  push<tchecker::integer_t>(left * right);
390  return top<tchecker::integer_t>();
391  }

394  case VM_MOD: {
395  auto const right = top_and_pop<tchecker::integer_t>();
396  auto const left = top_and_pop<tchecker::integer_t>();
397  push<tchecker::integer_t>(left % right);
398  return top<tchecker::integer_t>();
399  }
400 
401  // stack = v1 ... vK-2 (vK-1 != vK)
402  case VM_NE: {
403  auto const right = top_and_pop<tchecker::integer_t>();
404  auto const left = top_and_pop<tchecker::integer_t>();
405  push<tchecker::integer_t>(left != right);
406  return top<tchecker::integer_t>();
407  }
408 
409  // stack = v1 ... vK-2 (vK-1 + vK)
410  case VM_SUM: {
411  auto const right = top_and_pop<tchecker::integer_t>();
412  auto const left = top_and_pop<tchecker::integer_t>();
413  push<tchecker::integer_t>(left + right);
414  return top<tchecker::integer_t>();
415  }
416 
417  // stack = v1 ... vK-1 (- vK)
418  case VM_NEG: {
419  auto const v = top_and_pop<tchecker::integer_t>();
420  push<tchecker::integer_t>(-v);
421  return top<tchecker::integer_t>();
422  }
423 
424  // stack = v1 ... vK-1 (! vK)
425  case VM_LNOT: {
426  auto const v = top_and_pop<tchecker::integer_t>();
427  push<tchecker::integer_t>(!v);
428  return top<tchecker::integer_t>();
429  }
430 
431  // no-operation
432  case VM_NOP:
433  return 1;
434 
435  // stack = v1 ... vK-4 1 output (vK-2 vK-1 s vK) where s is a
436  // parameter of VM_CLKCONSTR (strictness)
437  case VM_CLKCONSTR: {
438  tchecker::bytecode_t cmp = *++bytecode;
439  auto const bound = top_and_pop<tchecker::integer_t>();
440  auto const id2 = top_and_pop<tchecker::clock_id_t>();
441  auto const id1 = top_and_pop<tchecker::clock_id_t>();
442  static_assert(tchecker::clock_constraint_t::LT == 0, "");
443  clkconstr.emplace_back(id1, id2, (cmp == 0 ? tchecker::clock_constraint_t::LT : tchecker::clock_constraint_t::LE), bound);
444  return 1;
445  }
446 
447  // stack = v1 ... vK-3 output (vK-2 vK-1 vK)
448  case VM_CLKRESET: {
449  auto const value = top_and_pop<tchecker::integer_t>();
450  auto const right_id = top_and_pop<tchecker::clock_id_t>();
451  auto const left_id = top_and_pop<tchecker::clock_id_t>();
452  clkreset.emplace_back(left_id, right_id, value);
453  return 1;
454  }

459  return 1;
