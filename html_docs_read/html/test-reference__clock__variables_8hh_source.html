74 TEST_CASE("Reference clock variables from system - no array", "[reference clock variables]")
75 {
76  std::string declarations = "system:access_map_no_clock_array \n\
77  event:a \n\
78  \n\
79  int:1:1:1:1:i \n\
80  clock:1:x \n\
81  clock:1:y \n\
82  \n\
83  process:P1 \n\
84  location:P1:l0{initial:} \n\
85  location:P1:l1{invariant: x<=1} \n\
86  edge:P1:l0:l1:a{provided: y>0 : do: x=0} \n\
87  \n\
88  process:P2 \n\
89  location:P2:l0{initial:} \n\
90  edge:P2:l0:l0:a{provided: i<=3} \n\
91  ";
92 
93  tchecker::parsing::system_declaration_t const * sysdecl = tchecker::test::parse(declarations);
94 
95  REQUIRE(sysdecl != nullptr);
96 
97  tchecker::ta::system_t system(*sysdecl);
98  tchecker::variable_access_map_t const vaccess_map = tchecker::variable_access(system);
99 
100  tchecker::process_id_t const P1 = system.process_id("P1");
101  tchecker::process_id_t const P2 = system.process_id("P2");
102  tchecker::clock_id_t const x = system.clock_id("x");
103  tchecker::clock_id_t const y = system.clock_id("y");
104 
105  tchecker::reference_clock_variables_t const reference_clocks =
106  tchecker::process_reference_clocks(vaccess_map, system.clock_variables().flattened(), system.processes_count());
107 
108  SECTION("Check reference clocks")
109  {
110  REQUIRE(reference_clocks.refcount() == system.processes_count());
111  REQUIRE(reference_clocks.refmap()[P1] == P1);
112  REQUIRE(reference_clocks.refmap()[P2] == P2);
113  REQUIRE(reference_clocks.procmap()[P1] == P1);
114  REQUIRE(reference_clocks.procmap()[P2] == P2);
115  }
116 
117  SECTION("Check clocks")
118  {
119  REQUIRE(reference_clocks.size() == reference_clocks.refcount() + 2); // x, y
120 
121  tchecker::clock_id_t xx = reference_clocks.refcount() + x;
122  tchecker::clock_id_t yy = reference_clocks.refcount() + y;
123 
124  REQUIRE(reference_clocks.translate_system_clock(x) == xx);
125  REQUIRE(reference_clocks.translate_system_clock(y) == yy);
126 
127  REQUIRE(reference_clocks.id("$x") == xx);
128  REQUIRE(reference_clocks.id("$y") == yy);
129 
130  REQUIRE(reference_clocks.refmap()[xx] == reference_clocks.refclock_of_system_clock(x));
131  REQUIRE(reference_clocks.refmap()[yy] == reference_clocks.refclock_of_system_clock(y));
132 
133  REQUIRE(reference_clocks.refmap()[xx] == P1);
134  REQUIRE(reference_clocks.refmap()[yy] == P1);
135  }
136  delete sysdecl;
137 }

228 TEST_CASE("translation of clock constraints", "[clocks]")
229 {
230  std::vector<std::string> refclocks = {"$0", "$1", "$2"};
231  tchecker::reference_clock_variables_t r(refclocks);
232  r.declare("x", "$0");
233  r.declare("y", "$1");
234  r.declare("z1", "$2");
235  r.declare("z2", "$2");
236 
237  tchecker::clock_id_t const t0 = r.id("$0");
238  tchecker::clock_id_t const t2 = r.id("$2");
239  tchecker::clock_id_t const x = r.id("x");
240  tchecker::clock_id_t const y = r.id("y");
241  tchecker::clock_id_t const z1 = r.id("z1");
242  tchecker::clock_id_t const z2 = r.id("z2");
243 
244  tchecker::clock_id_t const _x = 0;
245  tchecker::clock_id_t const _y = 1;
246  tchecker::clock_id_t const _z1 = 2;
247  tchecker::clock_id_t const _z2 = 3;
248 
249  SECTION("Upper-bound constraint")
250  {
251  tchecker::clock_constraint_t c =
252  r.translate(tchecker::clock_constraint_t(_x, tchecker::REFCLOCK_ID, tchecker::clock_constraint_t::LE, 1));
253  REQUIRE(c.id1() == x);
254  REQUIRE(c.id2() == t0);
255  REQUIRE(c.comparator() == tchecker::clock_constraint_t::LE);
256  REQUIRE(c.value() == 1);
257  }
258 
259  SECTION("Lower-bound constraint")
260  {
261  tchecker::clock_constraint_t c =
262  r.translate(tchecker::clock_constraint_t(tchecker::REFCLOCK_ID, _z1, tchecker::clock_constraint_t::LE, -3));
263  REQUIRE(c.id1() == t2);
264  REQUIRE(c.id2() == z1);
265  REQUIRE(c.comparator() == tchecker::clock_constraint_t::LE);
266  REQUIRE(c.value() == -3);
267  }
268 
269  SECTION("Diagonal constraint, same reference clock")
270  {
271  tchecker::clock_constraint_t c = r.translate(tchecker::clock_constraint_t(_z1, _z2, tchecker::clock_constraint_t::LT, 19));
272  REQUIRE(c.id1() == z1);
273  REQUIRE(c.id2() == z2);
274  REQUIRE(c.comparator() == tchecker::clock_constraint_t::LT);
275  REQUIRE(c.value() == 19);
276  }
277 
278  SECTION("Diagonal constraint, distinct reference clocks")

280  tchecker::clock_constraint_t c = r.translate(tchecker::clock_constraint_t(_y, _x, tchecker::clock_constraint_t::LE, -5));
