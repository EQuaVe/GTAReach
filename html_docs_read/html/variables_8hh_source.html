261 void declare_flattened_variable(ID id, std::string const & name, INFO const & info,
262  std::function<void(ID, std::string const &, INFO const &)> declare)
263 {
264  static_assert(std::is_base_of<tchecker::size_info_t, INFO>::value, "INFO should provide variable size");
265 
266  typename INFO::size_t size = info.size();
267 
268  if (size == 1)
269  declare(id, name, info);
270  else {
271  INFO flat_info{info};

273 
274  for (typename INFO::size_t i = 0; i < size; ++i) {
275  std::stringstream ss;
276  ss << name << "[" << i << "]";
277  declare(id + i, ss.str(), flat_info);

374  void declare(ID id, std::string const & name, INFO const & info)
375  {
376  if (id < _next_id)
377  throw std::invalid_argument("Variable identifier is already used");
